Enum estado_orden {
  pendiente
  pagada
  enviada
  entregada
  cancelada
}

Table Cliente {
  id int [pk, increment]
  nombre varchar(100) [not null]
  apellido varchar(100) [not null]
  email varchar(150) [not null, unique]
  telefono varchar(20)
  fecha_registro datetime [not null]
}

Table Producto {
  id int [pk, increment]
  sku varchar(50) [not null, unique]
  nombre varchar(150) [not null]
  descripcion text
  categoria varchar(100) [not null]
  precio decimal(10,2) [not null]
  stock int [not null, default: 0]
  activo boolean [not null, default: true]
  fecha_creacion datetime [not null]
}

Table Sucursal {
  id int [pk, increment]
  nombre varchar(120) [not null]
  codigo_sucursal varchar(30) [not null, unique]
  calle varchar(150) [not null]
  numero varchar(20)
  colonia varchar(100)
  ciudad varchar(100) [not null]
  estado varchar(100) [not null]
  pais varchar(100) [not null, default: 'México']
  codigo_postal varchar(10)
  latitud decimal(9,6)
  longitud decimal(9,6)
}

Table Orden {
  id int [pk, increment]
  cliente_id int [not null]
  sucursal_id int [not null]
  fecha_orden datetime [not null]
  estado estado_orden [not null, default: 'pendiente']
  metodo_pago varchar(50)
  total decimal(12,2) [not null]
}

Table DetalleOrden {
  id int [pk, increment]
  orden_id int [not null]
  producto_id int [not null]
  cantidad int [not null]
  precio_unitario decimal(10,2) [not null]
  subtotal decimal(12,2) [not null]
}

Ref: Orden.cliente_id > Cliente.id
Ref: Orden.sucursal_id > Sucursal.id
Ref: DetalleOrden.orden_id > Orden.id
Ref: DetalleOrden.producto_id > Producto.id
